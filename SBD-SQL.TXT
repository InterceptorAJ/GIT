CREATE TABLE "HR1"."AUTHORS" (  
  "ID_AUTHOR" NUMBER NOT NULL ENABLE, 
  "NAME" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
  "SURNAME" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
  "EMAIL" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
  "PHONE_NO" NUMBER NOT NULL ENABLE
   ) ;

   COMMENT ON COLUMN AUTHORS.ID_AUTHOR IS 'Primary key of Authors table' 
   ;
   COMMENT ON COLUMN AUTHORS.NAME IS 'Author name' 
   ;
   COMMENT ON COLUMN AUTHORS.SURNAME IS 'Author surname'
   ;
   COMMENT ON COLUMN AUTHORS.EMAIL IS 'Author email' 
   ;
   COMMENT ON COLUMN AUTHORS.PHONE_NO IS 'Author phone number' 
   ;

CREATE UNIQUE INDEX "HR1"."AUTHORS_PK" ON "HR1"."AUTHORS" ("ID_AUTHOR")
   ;

   CREATE TABLE "HR1"."IMAGES" (
   "ID_IMAGE" NUMBER NOT NULL ENABLE, 
   "FILE_NAME" VARCHAR2(50 BYTE) NOT NULL ENABLE, 
   "IMAGE_TAB" ORDIMAGE NOT NULL ENABLE
   ) ;

   COMMENT ON COLUMN IMAGES.ID_IMAGE IS 'Primary key of Images table' 
   ;
   COMMENT ON COLUMN IMAGES.FILE_NAME IS 'Image filename' 
   ;
   COMMENT ON COLUMN IMAGES.IMAGE_TAB IS 'Image tab'
   ;
 
CREATE UNIQUE INDEX "HR1"."IMAGES_PK" ON "HR1"."IMAGES" ("ID_IMAGE") 
   ;

CREATE TABLE "HR1"."INGREDIENTS" (
  "ID_INGREDIENT" NUMBER NOT NULL ENABLE, 
  "NAME" VARCHAR2(20 BYTE) NOT NULL ENABLE
   ) ;

   COMMENT ON COLUMN INGREDIENTS.ID_INGREDIENT IS 'Primary key of Ingredients table' 
   ;
   COMMENT ON COLUMN INGREDIENTS.NAME IS 'Name of ingredient' 
   ;

CREATE UNIQUE INDEX "HR1"."INGREDIENTS_PK" ON "HR1"."INGREDIENTS" ("ID_INGREDIENT") 
   ;

CREATE TABLE "HR1"."RECIPE" (
   "ID_RECIPE" NUMBER NOT NULL ENABLE, 
   "NAME" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
   "DESCRIPTION" VARCHAR2(20 BYTE) NOT NULL ENABLE
   ) ;

   COMMENT ON COLUMN RECIPE.ID_RECIPE IS 'Primary key of Recipe table' 
   ;
   COMMENT ON COLUMN RECIPE.NAME IS 'Name of recipe' 
   ;
   COMMENT ON COLUMN RECIPE.DESCRIPTION IS 'Description of recipe' 
   ;

CREATE UNIQUE INDEX "HR1"."RECIPE_PK" ON "HR1"."RECIPE" ("ID_RECIPE") 
   ;
